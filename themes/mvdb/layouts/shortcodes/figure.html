{{- $src := .Get "src" -}} {{- if not $src }}{{ errorf "figure shortcode requires src: %s" .Position }}{{ end -}} {{- $center := or (strings.HasSuffix $src "#center") (eq (.Get "align") "center") -}} {{- $clean := strings.TrimSuffix "#center" $src -}} {{- $match := strings.TrimPrefix "./" $clean -}} {{- $resource := .Page.Resources.GetMatch $match -}} {{- $url := $clean -}} {{- $width := .Get "width" -}} {{- $height := .Get "height" -}} {{- with $resource -}} {{- $url = .RelPermalink -}} {{- if not $width }}{{ $width = .Width }}{{ end -}} {{- if not $height }}{{ $height = .Height }}{{ end -}} {{- end -}} {{- $caption := .Get "caption" | default (.Get "title") -}} {{- $alt := .Get "alt" | default ($caption | plainify) -}}
